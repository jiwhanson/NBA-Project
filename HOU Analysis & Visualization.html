#For each team, let's consider the number of 3's attempted and made
#Select only the needed columns
team_shots = shot_logs[['TEAM', 'PTS_TYPE', 'FGM']]

#Get 3's attempted and 3's made
team_3_total = team_shots[team_shots['PTS_TYPE'] == 3][['TEAM', 'FGM']]
team_3_total = team_3_total.groupby('TEAM').agg(Attempted3 = ('FGM', 'count'), 
                                                Made3 = ('FGM', 'sum')).reset_index()
#Throw this onto a scatterplot
team_3_total.plot.scatter(x = 'Attempted3', y = 'Made3',
                          title = "Attempted 3's vs Made 3's for NBA Teams", color = 'grey')
plt.ylim(50, 800)
#Highlight GSW
plt.scatter(x = 1501, y = 600, color = 'DarkGoldenRod')
plt.annotate('GSW', (1501, 600), textcoords = 'offset points', xytext = (0, 10), ha = 'center')
#Highlight HOU
plt.scatter(x = 1569, y = 556, color = 'DarkRed')
plt.annotate('HOU', (1569, 556), textcoords = 'offset points', xytext = (0, -17), ha = 'center')
